from itertools import product, permutations

def f(x, y, z, w):  
    return (x and z or (not w)) and x or y  
  
for x1, x2, x3, x4, x5, x6 in product([0, 1], repeat=6):  
    t = (  
        (0, x1, 0, x2, 1),  
        (x3, 1, x4, 1, 0),  
        (1, 1, x5, x6, 0)  
    )  
    if len(t) == len(set(t)):  
        for p in permutations('xyzw', r=4):  
            if all(f(**dict(zip(p, line))) == line[-1] for line in t):  
                print(*p)

from itertools import product, permutations  
  
  
def f1(x, y, z, w):  
    return (x == y) and (w <= z)  
  
def f2(x, y, z, w):  
    return (x <= y) <= (w == z)  
  
for x1, x2, x3, x4, x5 in product([0, 1], repeat=5):  
    t1 = (  
        (1, x1, 1, 1, 1),  
        (0, 1, 0, x2, 1),  
        (x3, 0, 0, x4, 0)  
    )  
  
    t2 = (  
        (1, x1, 1, 1, 0),  
        (0, 1, 0, x2, x5),  
        (x3, 0, 0, x4, 0)  
    )  
  
    if len(t1) == len(set(t1)) and len(t2) == len(set(t2)):  
        for p in permutations('xyzw', r=4):  
            if all(f1(**dict(zip(p, line))) == line[-1] for line in t1) and all(f2(**dict(zip(p, line))) == line[-1] for line in t2):  
                print(*p)
