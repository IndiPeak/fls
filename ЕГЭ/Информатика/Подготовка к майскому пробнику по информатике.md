```python
A = [["x", "y", "z", "w"]]

                    A.append([x, y, z, w])

  
  

def r(v):

    return list(map(list, zip(*v[::-1])))

  
  

def q(a, b):

    Q = a[1:]

    for ia, ib in zip(Q, b):

        for ja, jb in zip(ia, ib):

            if ja is not None\

                    and jb is not None\

                    and ja != jb:

                return False

    return True

  
  

_ = None

    [_, 0, _, _],

    [1, 0, _, 0],

    [1, _, 0, 0]

]
```

```python
with open("26.txt") as f:

    S, N = map(int, f.readline().split())

    files = sorted(int(f.readline()) for _ in range(N))

  

# Вычисляем максимальное количество файлов, которые можно сохранить

total_size = 0

count = 0

for size in files:

    if total_size + size <= S:

        total_size += size

        count += 1

    else:

        break

  

# Находим максимальный файл, который можно заменить

valid_files = [f for f in files if total_size - files[count - 1] + f <= S]

max_file = max(valid_files) if valid_files else 0

  

print(count, max_file)
```
