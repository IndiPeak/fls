```python
def f(h, p):  
    if h >= 42: return p % 2 == 0  
    if p == 0: return 0  
    n = [f(h+1, p-1), f(h+3, p-1), f(h*2, p-1)]  
    return any(n) if p % 2 != 0 else all(n)
```

```python
# 19
print(min([s for s in range(1, 42) if f(s, 2)]))

# 20
print([s for s in range(1, 42) if not f(s, 1)] and f(s, 3))

# 21
print(min([s for s in range(1, 42) if not f(s, 2) and f(s, 4)]))
```
